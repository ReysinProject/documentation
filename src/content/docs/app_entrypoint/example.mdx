### 1. Adding Global Styles

You can add global styles or CSS resets:

```typescript jsx
import { type AppContainer, Reysin, ReysinApp } from "@reysin/project";
import React from "react";
import "./global-styles.css";

new Reysin((container: AppContainer) => {
  return (
   <React.StrictMode>
    <ReysinApp container={container} />
   </React.StrictMode>
  );
});
```

### 2. Integrating with State Management Libraries

While Reysin provides its own state management, you can integrate with other libraries:

```typescript jsx
import { type AppContainer, Reysin, ReysinApp } from "@reysin/project";
import React from "react";
import { Provider } from "react-redux";
import store from "./store";

new Reysin((container: AppContainer) => {
  return (
   <Provider store={store}>
     <React.StrictMode>
      <ReysinApp container={container} />
     </React.StrictMode>
   </Provider>
  );
});
```

### 3. Adding Error Boundaries

Implement error boundaries for better error handling:

```typescript jsx
import { type AppContainer, Reysin, ReysinApp } from "@reysin/project";
import React from "react";
import ErrorBoundary from "./ErrorBoundary";

new Reysin((container: AppContainer) => {
  return (
   <ErrorBoundary>
     <React.StrictMode>
      <ReysinApp container={container} />
     </React.StrictMode>
   </ErrorBoundary>
  );
});
```

### 4. Adding Internationalization

Integrate internationalization libraries:

```typescript jsx
import { type AppContainer, Reysin, ReysinApp } from "@reysin/project";
import React from "react";
import { I18nextProvider } from "react-i18next";
import i18n from "./i18n";

new Reysin((container: AppContainer) => {
  return (
   <I18nextProvider i18n={i18n}>
     <React.StrictMode>
      <ReysinApp container={container} />
     </React.StrictMode>
   </I18nextProvider>
  );
});
```
